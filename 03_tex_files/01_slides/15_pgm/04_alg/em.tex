\begin{algorithm}[H]
	\setstretch{1.0}
	\DontPrintSemicolon
	\footnotesize
	initialize parameters $\bm{\theta}$\;
	\While{not converged}{
		compute pseudo counts\;
		set parameters to the maximum likelihood estimates\;
	}
	
	\Return{final parameters $\bm{\theta}$}
 	\caption{Expectation-Maximization Algorithm}
\end{algorithm}
