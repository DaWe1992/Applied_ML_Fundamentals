# This declares a variable s initialized with value "Hello"
s = 'Hello'

# You may also use double quotes
s = "Hello"

# Indexing is possible (starts at 0)
s = "Hello"[1] # gives "e"

# Indexing from the back
s = "Hello"[-1] # gives "o"

# Dynamic typing: Now you can use the same variable to save numbers,
# but keep track of what you do with your variables!
s = 0.50**2 * 3.1415
